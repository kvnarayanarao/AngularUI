        function($route, $rootScope, config, requestHandlers, filterService, Crossfilter) {
 
            var filtersPeriodPopulation = function($scope) {

                $scope.counter = 1;

            function filterData(filterData, $scope) {
 
                //Department
                filterService.data.departmentData = _.sortBy(_.unique(filterData.collection(), 'd'), 'd');
                filterService.data.mdlDepartment = filterService.data.departmentData[0];
                //Category Group
                filterData.filterBy('d', filterService.data.departmentData[0].d);
                filterService.data.categoryGroupData = _.sortBy(_.unique(filterData.collection(), 'cg'), 'cg');
                filterService.data.mdlCategoryGroup = filterService.data.categoryGroupData[0];
                //Category
                filterData.filterBy('cg', filterService.data.categoryGroupData[0].cg);
                filterService.data.categoryData = _.sortBy(_.unique(filterData.collection(), 'c'), 'c');
                filterService.data.categoryData.unshift({ 'c': "ALL" });
                filterService.data.mdlCategory = filterService.data.categoryData[0];
                //Sub Category
                filterService.data.subCategoryData = [];
                filterService.data.subCategoryData.unshift({ 'sc': "ALL" });
                filterService.data.mdlSubCategory = filterService.data.subCategoryData[0];
                //Segment
                filterService.data.segmentData = [];
                filterService.data.segmentData.unshift({ 's': "ALL" });
                filterService.data.mdlSegment = filterService.data.segmentData[0];
 
                if ($route.current.loadedTemplateUrl.match('assortmentCoverage')) {
                    renderDMAData($scope);
                } else if ($route.current.loadedTemplateUrl.match('brandProfile')) {
 
                    filterService.data.mdlBrandHierList = filterService.data.brandHierList[0];
                    $scope.loadingFilter = false;
                    $scope.loadingImage = false;
                    if (filterService.data.mdlBrandType == "ALL" || filterService.data.mdlBrandHierList == "ALL")
                        $scope.btnFilterApply = true;
                    else
                        $scope.btnFilterApply = false;
                }
 
            }

                if (filterService.data.mdlCategory.c == "ALL")
                    filterService.data.dmaData.filterBy('id', filterService.data.mdlCategoryGroup.id);

                    filterService.data.dmaData.filterBy('id', filterService.data.mdlCategory.id);
                else if (filterService.data.mdlCategory.c != "ALL" && filterService.data.mdlSubCategory.sc != "ALL")
                    filterService.data.dmaData.filterBy('id', filterService.data.mdlSubCategory.id);

                    filterService.data.dmaData.filterBy('id', filterService.data.mdlSegment.id);
