 (function() {
     'use strict';
 
     angular.module("ABC").controller("TabController", ['$scope', '$rootScope', 'chartHandlerService', '$route', '$routeParams', 'filterService',
         function($scope, $rootScope, chartHandlerService, $route, $routeParams, filterService) {
 
             $rootScope.serverError = false;
 
             $scope.tabs = [{
                 id: 1,
                 desc: "Assortment Coverage",
                 link: "#/assortmentCoverage"
             }, {
                 id: 2,
                 desc: "Brand Profile",
                 link: "#/brandProfile"
             }, {
                 id: 3,
                 desc: "Buyer Scope",
                 link: "#/buyerScope"
             }];
 
             $scope.currentTab = 1;
 
             $scope.$on("$routeChangeSuccess", function(event, next, current) {
                 switch (next.originalPath) {
                     case "/assortmentCoverage":
                         $scope.currentTab = 1;
                         break;
                     case "/brandProfile":
                         $scope.currentTab = 2;
                         break;
                     case "/buyerScope":
                         $scope.currentTab = 3;
                         break;
                     default:
                         $scope.currentTab = 1;
                         break;
                 };
 
                 $rootScope.isSnapShotEnabled = true;
                 $rootScope.isExportButtonEnabled = true;
             });
 
             this.alertClose = function() {
                 $rootScope.serverError = false;
             }
 
             this.setCurrentTab = function(currentTab) {
                 $scope.currentTab = currentTab.id;
                 var svg = d3.select("svg");
                 svg.selectAll("*").remove();
                 jQuery("div.col-md-9 h1").html('');
                 filterService.showCategoryLbl = "";
                 if (filterService.hasOwnProperty('mdlPeriod')) {
                     delete filterService['mdlPeriod'];
                 }
                 jQuery('.d3-tip').hide();
             };
 
 
             this.takeSnapshot = function() {
                 chartHandlerService.takeSnapshot(document.getElementsByTagName("svg"));
             };
 
             this.exportToCSV = function() {
                 chartHandlerService.exportToCSV();
             };
 
         }
     ]);
 })();
