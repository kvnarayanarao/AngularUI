(function() {
    var moduleHandlers = angular.module("ABC");
 
    moduleHandlers.factory("filterChangeHandler", ['$route', '$rootScope', 'config', 'requestHandlers', 'filterService', 'Crossfilter',
        function($route, $rootScope, config, requestHandlers, filterService, Crossfilter) {
 
            var filtersDataPopulation = function($scope) {
                if (!filterService.data.isSet) {
                    requestHandlers.requestAjaxService('filters', "{}")
                        .then(
                            function(successResponse) {
                                if (successResponse.status == 200 && successResponse.data != "") {
                                    filterService.data.filterData = new Crossfilter(successResponse.data.records, 'id', 'persistent');
                                    dataRender(filterService.data.filterData, $scope);
                                    filterService.data.isSet = true;
                                } else {
                                    requestHandlers.erorHandler(false, $scope);
                                }
                            },
                            function(errorResponse) {
                                requestHandlers.erorHandler(errorResponse, $scope);
                            });
 
                } else {
                    dataRender(filterService.data.filterData, $scope);
                }
 
            }
 
            function dataRender(filterData, $scope) {
 
                filterService.data.periodData = _.sortBy(_.unique(filterData.collection(), 'period'), 'period');
                filterService.data.mdlPeriod = filterService.data.periodData[0];
 
                filterData.filterBy('period', filterService.data.periodData[0].period);
                filterService.data.departmentData = _.sortBy(_.unique(filterData.collection(), 'department'), 'department');
                filterService.data.mdlDepartment = filterService.data.departmentData[0];
 
                filterData.filterBy('department', filterService.data.departmentData[0].department);
                filterService.data.categoryGroupData = _.sortBy(_.unique(filterData.collection(), 'category_group'), 'category_group');
                filterService.data.mdlCategoryGroup = filterService.data.categoryGroupData[0];
 
                filterData.filterBy('category_group', filterService.data.categoryGroupData[0].category_group);
                filterService.data.categoryData = _.sortBy(_.unique(filterData.collection(), 'category'), 'category');
                filterService.data.categoryData.unshift({ 'category': "ALL" });
                filterService.data.mdlCategory = filterService.data.categoryData[0];
 
                filterService.data.subCategoryData = [];
                filterService.data.subCategoryData.unshift({ 'subcategory': "ALL" });
                filterService.data.mdlSubCategory = filterService.data.subCategoryData[0];
 
                if ($route.current.loadedTemplateUrl.match('assortmentCoverage')) {
                    renderDMAData($scope);
                } else if ($route.current.loadedTemplateUrl.match('brandProfile')) {
                    renderBrands($scope);
                }
            }
 
            function renderDMAData($scope) {
                if (!filterService.data.isDMASet) {
                    requestHandlers.requestMapService('app/data/dma.json').then(
                        function(successResponse) {
                            if (successResponse.status == 200 && successResponse.data != "") {
                                filterService.data.dmaNames = new Crossfilter(successResponse.data.map, 'dma_id', 'persistent');
                                filterService.data.dmaData = new Crossfilter(successResponse.data.data, 'id', 'persistent');
                                dmaRender($scope);
                                filterService.data.isDMASet = true;
                            } else {
                                requestHandlers.erorHandler(false, $scope);
                                // $scope.counter = 0;
                                // disableLoading($scope);
                                // $rootScope.serverError = true;
                                // $rootScope.serverErrorText = config.serverErrorMessage;
                            }
                        },
                        function(errorResponse) {
                            requestHandlers.erorHandler(errorResponse, $scope);
                            return false;
                        });
                } else {
                    dmaRender($scope);
                }
            }
 
            function renderBrands($scope) {
                var parameters = {};
                parameters['period'] = "'" + filterService.data.mdlPeriod.period + "'";
                parameters['department'] = "'" + filterService.data.mdlDepartment.department + "'";
                parameters['categoryGroup'] = "'" + filterService.data.mdlCategoryGroup.category_group + "'";
                if (filterService.data.mdlCategory.category != "ALL") {
                    parameters['category'] = "'" + filterService.data.mdlCategory.category + "'";
                }
                if (filterService.data.mdlSubCategory.subcategory != "ALL") {
                    parameters['subcategory'] = "'" + filterService.data.mdlSubCategory.subcategory + "'";
                }
                parameters['brandHierarchy'] = filterService.data.mdlBrandHier.val;
                if (filterService.data.controlBrands) {
                    parameters['controlBrand'] = "'" + "CTL BR" + "'";
                }
                if (filterService.data.clubPacks) {
                    parameters['clubPack'] = "'" + "Y" + "'";
                }
                if (filterService.data.mdlBrandType != "ALL")
                    parameters['brandClass'] = "'" + filterService.data.mdlBrandType + "'";
 
 
                requestHandlers.requestAjaxService('brandType', parameters)
                    .then(
                        function(successResponse) {
                            if (successResponse.status == 200 && successResponse.data != "") {
                                parseData(successResponse, $scope, 6, 0);
                                $scope.counter--;
                                disableLoading($scope);
                                angular.copy(parameters, $scope.paramForRowExpand);
                                if (filterService.data.mdlBrandType == "ALL" || filterService.data.mdlBrandHierList == "ALL")
                                    $scope.btnFilterApply = true;
                                else
                                    $scope.btnFilterApply = false;
 
                                filterService.data.isSet = true;
                            } else {
                                requestHandlers.erorHandler(false, $scope);
                            }
                        },
                        function(errorResponse) {
                            $scope.filterError.brandNames = 1
                            requestHandlers.erorHandler(errorResponse, $scope);
                            return false;
                        });
 
 
            }
 
            function dmaRender($scope) {
                filterService.data.dmaNames.unfilterAll();
                filterService.data.dmaData.unfilterAll();
 
                if (filterService.data.mdlCategory.category == "ALL")
                    filterService.data.dmaData.filterBy('id', filterService.data.mdlCategoryGroup.i);
                else if (filterService.data.mdlCategory.category != "ALL" && filterService.data.mdlSubCategory.subcategory == "ALL")
                    filterService.data.dmaData.filterBy('id', filterService.data.mdlCategory.i);
                else if (filterService.data.mdlCategory.category != "ALL" && filterService.data.mdlSubCategory.subcategory != "ALL")
                    filterService.data.dmaData.filterBy('id', filterService.data.mdlSubCategory.i);
 
                var _tempDma = filterService.data.dmaData.collection();
                if (_tempDma.length && _tempDma[0].dma_ids) {
                    filterService.data.dmaNames.filterBy('dma_id', _tempDma[0].dma_ids, function(i, d) { return (_.contains(i, d)) ? i : false; });
                    filterService.data.totalGeography = _.sortBy(_.unique(filterService.data.dmaNames.collection(), 'dma'), 'dma');
                    filterService.data.totalGeography.unshift({ "dma": "NATIONAL" });
                    filterService.data.mdlGeo = filterService.data.totalGeography[0];
                }
                disableLoading($scope);
            }
 
            function disableLoading($scope) {
                $scope.$watch("[ChartTableLoadingTimer.chart,ChartTableLoadingTimer.table,counter]", function(newValue, oldValue) {
                    if (newValue[0] == false && newValue[1] == false && newValue[2] <= 0) {
                        $scope.loadingFilter = false;
                        $scope.loadingImage = false;
                        $scope.counter = 0;
                    }
                });
            }

            function filterChange(selectedFilter, selectedValue, $scope) {
 
                var filterParam = {};
                if (selectedFilter == 'Period' ||

                    selectedFilter == 'Category' ||
                    selectedFilter == 'SubCat' ||

                    filterService.data.totalGeography = [{ "dma": "NATIONAL" }];
                    filterService.data.mdlBrandType = "ALL";

                    filterService.data.mdlCategoryGroup = filterService.data.categoryGroupData[0];
 
                    filterService.data.filterData.filterBy('category_group', filterService.data.mdlCategoryGroup.category_group);

                    $scope.disSubCat = true;

                    filterService.data.filterData.filterBy('department', selectedValue.department);

                    filterService.data.mdlSubCategory = filterService.data.subCategoryData[0];
                    $scope.disSubCat = true;
                } else if (selectedFilter == 'CategoryGroup') {
 
                    filterService.data.filterData.filterBy('category_group', selectedValue.category_group);
                    filterService.data.categoryData = _.sortBy(_.unique(filterService.data.filterData.collection(), 'category'), 'category');
                    filterService.data.categoryData.unshift({ 'category': "ALL" });
                    filterService.data.mdlCategory = filterService.data.categoryData[0];
                    filterService.data.subCategoryData.unshift({ 'subcategory': "ALL" });
                    filterService.data.mdlSubCategory = filterService.data.subCategoryData[0];
                    $scope.disSubCat = true;
                } else if (selectedFilter == 'Category') {
                    if (filterService.data.mdlCategory.category != "ALL") {
                        filterService.data.filterData.filterBy('category', filterService.data.mdlCategory.category);
                        filterService.data.subCategoryData = _.sortBy(_.unique(filterService.data.filterData.collection(), 'sc'), 'sc');
                        filterService.data.subCategoryData.unshift({ 'subcategory': "ALL" });
                        filterService.data.mdlSubCategory = filterService.data.subCategoryData[0];

                    filterService.data.mdlSubCategory = selectedValue;

                            filterService.data.mdlBrandType = value.type;

            var parseData = function(response, $scope, orderLevel, onLoadSettings) {
                if (orderLevel == 6) {
                    var brandRecords = response.data.records;
                    var length = response.data.numberOfRecords;
                    $scope.disbrandHL = true;
                    if (length == 0) {
                        $scope.filterError.brandNames = 1;
                        $scope.errorModalWindow = true;
                        return false;
                    }
                    filterService.data.TotalBrands = [];
                    var totalBrandNames = [];
                    for (i = 0; i < length; i++) {
                        var rowData = brandRecords[i];
                        totalBrandNames.push(rowData['brand']);
                        var TotalBrands = { brand: rowData['brand'], type: rowData['Brand Class'] }
                        filterService.data.TotalBrands.push(TotalBrands);
                    }
                    filterService.data.brandHierList = totalBrandNames.sort();
                    filterService.data.brandHierList.unshift("ALL");
                    $scope.responseBody["BrandHierarchyList"] = "'" + filterService.data.brandHierList[1] + "'";
                    if ($scope.pageName = "brandProfileTab") {
                        filterService.data.mdlBrandHierList = filterService.data.brandHierList[0];
                        filterService.data.mdlBrandType = "ALL";
                        $scope.disbrandHL = false;
                    }
                }
                return $scope.responseBody;
            };
