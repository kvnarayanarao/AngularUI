(function() {
    var app = angular.module("ABC");
 
    app.factory("requestHandlers", ['$http', '$rootScope', 'config', function($http, $rootScope, config) {
        var urlProvider = function() {
            var returnOBJ = {
                "serviceURL": "",
                "exportURL": ""
            };
 
            returnOBJ['serviceURL'] = window.location.href.split('/')[2];
            switch (returnOBJ['serviceURL']) {
                case config.prod.serviceURL:
                    returnOBJ['exportURL'] = config.prod.exportURL;
                    break;
                default:
                    returnOBJ['serviceURL'] = config.dev.serviceURL;
                    returnOBJ['exportURL'] = config.dev.exportURL;
                    break;
            }
            return returnOBJ;
        }
 
        var requestAjaxService = function(operationToSend, paramToSend) {
 
            var filterURL = urlProvider();
            return $http({
                url: 'http://' + filterURL['serviceURL'] + operationToSend,
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                    'Content-Encoding': 'gzip'
                },
                data: paramToSend
            });
        }
 
        var requestAjaxServiceExport = function(operationToSend, paramToSend) {
 
            var filterURL = urlProvider();
            return $http({
                url: 'http://' + filterURL['exportURL'] + '/data-service-web/webapi/dataRetrievalService/runOperation',
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                    'Content-Encoding': 'gzip'
                },
                data: {
                    "operation": operationToSend,
                    "params": paramToSend
                }
            });
        }
 
        var requestMapService = function(url) {
            return $http({
                url: url,
                method: 'GET',
                headers: {
                    'Content-Type': 'application/json',
                    'Content-Encoding': 'gzip'
                },
                cache: true,
                crossDomain: true
 
            });
        }
 
        var redirectWhenCookieisNot = function() {
            var sessionID = $cookies.get('sessionId');
            if (sessionID == '' || !angular.isDefined(sessionID) || sessionID == null || sessionID == 'null') {
                $rootScope.loginCheck = true;
            }
            return false;
        }
        var erorHandler = function(errorMessage, $scope) {
            if (errorMessage != false && errorMessage.status != 200)
                console.log(errorMessage);
            $scope.counter = 0;
            $scope.loadingFilter = false;
            $scope.loadingImage = false;
            $scope.loadingSym = false;
            $scope.loadingSym1 = false;
            $scope.loadingSym2 = false;
            $scope.loadingFilter = false;
            $rootScope.serverError = true;
            $rootScope.serverErrorText = config.serverErrorMessage;
            return false;
        }
 
        return {
            requestAjaxService: requestAjaxService,
            requestAjaxServiceExport: requestAjaxServiceExport,
            requestMapService: requestMapService,
            erorHandler: erorHandler
        };
    }]);
 
})();
